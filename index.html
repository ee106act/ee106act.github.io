#part f (retrieval)
def position_from_state(thetas):
    x = [(Tx + L*math.sin(theta)) for theta in thetas]
    y = [(Ty - L*math.cos(theta)) for theta in thetas]
    return x, y

def state_eq(X,t, nonzerotorque = False): #generalize with torque GENERALIZED FOR PART I TO INCLUDE VARIABLE TORQUES
    if nonzerotorque:
        tau = -M*R**2*(alpha*(X[0]-theta_d) + beta*X[1])
    else:
        tau = 0
    
    return [X[1], (-g*math.sin(X[0]))/R + tau/(M*R**2)]
